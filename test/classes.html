Hide / Show classes

.hide-phone :: hide if screen is < 480px

.hide-tablet :: hide if screen is < 767px

.hide-tab-x :: hide if screen is < 1023px

.hide-desktop :: hide if screen is < 1599px

.hide-big-desktop :: hide if screen is < 1919px

.show-phone :: show if screen is < 480px
.show-tablet :: show if screen is > 480px and < 767px
.show-tab-x :: show if screen is > 768px and < 1023px
.show-desktop :: show if screen is > 1024px and < 1599px
.show-big-desktop :: show if screen > 1599
